** mocha / describe,it,assert

- test/
- --exit
- .mocharc.json

#+begin_src js
  before((done) => {});
  after((done) => {});
  describe('Array', () => {
    describe('#indexOf()', () => {
      it('should return -1 the values is not present', () => {
        assert.equal([1,2,3].indexOf(4),-1);
      });
    });
  });
#+end_src
